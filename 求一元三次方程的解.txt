#include<stdio.h>
#include<math.h>
int a,b,c,d,i;
double x,x1,x2;
double FC(double x){
	return pow(x,3)*a+b*pow(x,2)+c*x+d;
}
int main(){
	scanf("%d %d %d %d",&a,&b,&c,&d);
	for(i=-100;i<=100;i++){
		x1=i;
		x2=i+1;
		if(FC(i) == 0){
			printf("%.2f ",x1);
		}
		//如果不写以下逼近方法,会有极小值无法输出,比如5e-12,i自增1,5.00就永远无法输出
		else{
			if(FC(x1)*FC(x2)<0){
				//二分法逼近x
				while(x2-x1>=0.001){
					x=(x1+x2)/2;
					if(FC(x1)*FC(x)<=0) x2=x;
					else x1=x;
				}
				printf("%.2f",x1);
			}
		}
	}
	return 0;
}
