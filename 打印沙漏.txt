#include<iostream>
using namespace std;
int sha(int n){
	if(n==1) return 1;
	else return 2*n-1+sha(n-1);
}
int main(){
    int N,n;
    string s;
    cin>>N;
    cin>>s;
    if(N==1) cout<<s<<"0";
    for(int i=1;i<N;i++){
    	if(2*sha(i)-1<=N && 2*sha(i+1)-1>=N) n=i;
	}
	for(int i=n;i>=1;i--){
        for(int j=1;j<=n-i;j++){
            cout<<" ";
        }
        for(int k=2*i-1;k>=1;k--){
            cout<<s;
        }
        cout<<endl;
    }
    for(int i=2;i<=n;i++){
        for(int j=1;j<=n-i;j++){
            cout<<" ";
        }
        for(int k=1;k<=2*i-1;k++){
            cout<<s;
        }
        cout<<endl;
    }
    cout<<N-sha(n)*2+1;
    return 0;
}
