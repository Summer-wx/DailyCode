#include<stdio.h>
#include<string.h>
int main(){
	int count=0,i,j,len,k;
	char S[1000],c1,c2;
	scanf("%d",&k);
	scanf("%s %c %c",&S,&c1,&c2);
	len=strlen(S);
	for(i=0;i<len-k+1;i++){
		for(j=i+k-1;j<len;j++){
			if(S[i]==c1 && S[j]==c2){
				count++;
			}
		}
	}
	printf("%d",count);
	return 0;
}
