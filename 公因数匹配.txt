#include<stdio.h>
int gys(int a,int b){
	int count=0;
	for(int i=b;i>1;i--){
		if(a%i==0 && b%i==0) count++;
	}
	if(count==0) return 0;
	else return 1;
}

int main(){
	int n,dig[1000],i,j;
	int p[1000][2],m=0;
	scanf("%d",&n);
	for(i=1;i<=n;i++){
		scanf("%d",&dig[i]);
	}
	for(i=1;i<n;i++){
		for(j=i+1;j<=n;j++){
			if(gys(dig[i],dig[j]) == 1){
				p[m][0]=i;
				p[m][1]=j;
				m++;
			}
		}
	}
	
	int min_1=9,min_2=9;
	for(i=0;i<m;i++){
		if(p[i][0]<min_1) min_1=p[i][0];
	}
	for(i=0;i<m;i++){
		if(p[i][0]==min_1 && p[i][1]<min_2) min_2=p[i][1];
	}
	printf("%d %d",min_1,min_2);
	return 0;
}
