#include<stdio.h>
#include<string.h>
//n个人存在k个关系 
int main(){
	int n,k,i,j;
	int rel[101][3];
	int f[100][100];
	
	memset(f,0,sizeof(f));
	memset(rel,0,sizeof(rel));
	scanf("%d %d",&n,&k);
	for(i=1;i<=k;i++){
		for(j=1;j<=2;j++){
			scanf("%d",&rel[i][j]);
		}
	}
	for(i=0;i<n;i++){
		f[i][0]=i+1;
	}
	
	for(i=1;i<=n;i++){
		int x=1;
		for(j=1;j<=k;j++){
			if(rel[j][1]==i){
				f[i-1][x]=rel[j][2];
				f[x+i-1][0]=0;
				x++;
			}
		}
	}
	
	int count=0,max=0;
	for(i=0;i<n;i++){
		if(f[i][0]!=0) count++;
		int sum=0;
		for(j=0;j<n;j++){
			if(f[i][j]!=0) sum++;
		}
		if(sum>max) max=sum;
	}
	
	printf("%d %d",count,max);
	return 0;
}
