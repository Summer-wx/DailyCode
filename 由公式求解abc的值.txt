/*一个数组中有a个1，b个2，c个3，如P(1,2)=ab/C(a+b+c,2),c为排列，在abc中任取两个数
输出abc的值为多少，已知P(1,2)=517/2091,P(2,3)=2632/10455,P(1,3)=308/2091*/ 
#include<stdio.h>
//定义一个求最大公约数的函数
int gys(int a,int b){
	int temp;
	while(b){
		temp=b;
		b=a%b;
		a=temp;
	}
	return a;
} 
//a:b:c=s1s3:s1s2:s2s3 
int main(){
	int s1=517*5,s2=2632,s3=308*5;
	int b=gys(s1*s3,gys(s1*s2,s2*s3));
	printf("%d,%d,%d",s1*s3/b,s1*s2/b,s2*s3/b);
}
