#include<stdio.h>
int isFB(int n){
	int a[50],i;
	a[0]=n/1000000;
	a[1]=n/100000%10;
	a[2]=n/10000%10;
	a[3]=n/1000%10;
	a[4]=n/100%10;
	a[5]=n/10%10;
	a[6]=n%10;
	for(i=7;i<50;i++){
		a[i]=a[i-1]+a[i-2]+a[i-3]+a[i-4]+a[i-5]+a[i-6]+a[i-7];
		if(a[i]==n) return 1;
		if(a[i]>n) return 0;
	}
	return 0;	
}

int main(){
	int i;
	for(i=9999999;i>=1000000;i--){
		if(isFB(i)) {
			printf("%d",i);
			break;
		}
	}
	return 0;
}
