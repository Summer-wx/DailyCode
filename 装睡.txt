#include<stdio.h>
char name[10][10];
int hx[10],mb[10];
int main(){
	int i,n,n1,k=0;
	scanf("%d",&n);
	n1=n;
	while(n1--){
		scanf("%s %d %d",&name[k],&hx[k],&mb[k]);
		k++;
	}
	for(i=0;i<n;i++){
		if(hx[i]<15 || hx[i]>20 || mb[i]<50 || mb[i]>70){
			printf("%s",name[i]);
			if(i!=n-1) printf("\n");
		}	
	}
    return 0;
}
