#include<stdio.h>
#include<string.h>
int main(){
	//char s[]="asdasdf";定义字符串
	//strcmp(a,b) 判断a，b是否相等，a=b则返回0，a>b则返回正数，反之返回负数
	//if(a==b)只是判断ab的地址是否相等
	int D,i,j,x[1000],k=0;
	int count;
	scanf("%d",&D);
	for(i=1;i<=D;i++){
		char a[1000],b[1000];
		scanf("%s",&a);
		scanf("%s",&b);
		if(strcmp(a,b)==0) x[k]=0;
		else if(a[0]!=b[0] || a[strlen(a)-1]!=b[strlen(a)-1]) x[k]=-1;
		else {
			count=0;			
			for(j=1;j<strlen(a)-1;j++){
				if(a[j]!=b[j]){
					if(b[j-1]!=b[j] && b[j+1]!=b[j] && b[j-1]==b[j+1]){
						count++;
						b[j]=a[j];
					}
				}
			}
			if(strcmp(a,b)==0) x[k]=count;
			else x[k]=-1;
		}
		k++;
	}
	for(i=0;i<D;i++){
		printf("%d\n",x[i]);
	}
	return 0;
}
