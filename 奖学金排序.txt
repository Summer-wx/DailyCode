#include<stdio.h>
int main(){
	int n,i,j,sum,k=1;
	scanf("%d",&n);
	int a[n][3]={0},b[n][2]={0};
	for(i=0;i<n;i++){
		sum=0;
		b[i][0]=k;
		k++;
		for(j=0;j<3;j++){
			scanf("%d",&a[i][j]);
			sum+=a[i][j];
		}
		b[i][1]=sum;
	}

	int t1,t2;
	for(i=0;i<n-1;i++){
		for(j=i+1;j<n;j++){
			if(b[i][1]<b[j][1]){
				t1=b[i][1];
				b[i][1]=b[j][1];
				b[j][1]=t1;
				t2=b[i][0];
				b[i][0]=b[j][0];
				b[j][0]=t2;
			}
		}
	}
	for(i=0;i<5;i++){
		for(j=i+1;j<5;j++){
			if(b[i][1]==b[j][1]){
				if(a[b[i][0]][0]<a[b[j][0]][0]){
					t1=b[i][1];
					b[i][1]=b[j][1];
					b[j][1]=t1;
					t2=b[i][0];
					b[i][0]=b[j][0];
					b[j][0]=t2;					
				}
				if(a[b[i][0]][0]==a[b[j][0]][0]){
					if(b[i][0]>b[j][0]){
						t1=b[i][1];
						b[i][1]=b[j][1];
						b[j][1]=t1;
						t2=b[i][0];
						b[i][0]=b[j][0];
						b[j][0]=t2;							
					}
				}
			}
		}
	} 	
	for(i=0;i<5;i++){
		for(j=0;j<2;j++){
			printf("%d ",b[i][j]);
		}
		printf("\n");
	} 
	return 0;
}
