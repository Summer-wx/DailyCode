#include<stdio.h>
#include<string.h>
int m,n,sum=0; 
int a[100][30];
char s[200];
int main(){
	int i,j;
	scanf("%d %d",&n,&m);
	for(i=0;i<n;i++){
		for(j=0;j<m;j++){
			scanf("%d",&a[i][j]);	
		}
	}
	scanf("%s",&s);
	for(i=1;i<strlen(s)-1;i+=3){
		int row=s[i+1]-49,col=s[i]-65;
		sum+=a[row][col];
	}
	printf("%d",sum);
	return 0;
} 
/*样例
3 5
9 3 13 8 12
0 32 1 3 4
8 7 12 34 1
=A1+A2+C3+A1+E3
样例输出 31
说明： A1+A2+C3+A1+E3=9+0+12+9+1=31
1<=n<=100 1<=m<=26 0<=数据<=10000
n*m的矩阵
