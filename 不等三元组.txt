//在数列a中，对于下标满足0<i<j<k<n+1且三个数不相同,则称i,j,k为一组不等三元组
//给定一个数组，问序列中有多少个不等三元组
//以下代码只能求解n<1000的数 
#include<stdio.h>
#define N 100005
long long n,count=0;
long long a[N];
long long check[N];
void dfs(int depth,int start){
	int i;
	if(depth==3){
		if(check[0]!=check[1]&&check[2]!=check[1]&&check[0]!=check[2]){
			count++;
		}
		return ;
	}
	for(i=start;i<n;i++){
		check[depth]=a[i];
		dfs(depth+1,i+1);
	}
}
int main(){
	int i;
	scanf("%lld",&n);
	for(i=0;i<n;i++){
		scanf("%lld",&a[i]);
	}
	dfs(0,0);
	printf("%lld\n",count); 
	return 0;
} 
/*
5
2 5 3 5 2
输出4
*/
