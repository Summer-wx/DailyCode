#include<stdio.h>
//输出1-n的所有子集 
int n;
int st[150];//0代表待选，1代表未选，2代表已选 
void dfs(int x){
	if(x>n){
		int i;
		for(i=1;i<=n;i++){
			if(st[i] == 1) printf("%d",i);
		}
		printf("\n");
		return ; 
	}
	//选 
	st[x]=1;
	dfs(x+1);
	//不选
	st[x]=2;
	dfs(x+1);
}
int main(){
	scanf("%d",&n);
	dfs(1);
	return 0;
}
