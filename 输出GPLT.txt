#include<stdio.h>
#include<string.h>
/*给定一个长度不超过10000的、仅由英文字母构成的字符串。请将字符重新调整顺序，按GPLTGPLT....这样的顺序输出，并忽略其它字符。当然，四种字符（不区分大小写）的个数不一定是一样多的，若某种字符已经输出完，则余下的字符仍按GPLT的顺序打印，直到所有字符都被输出。*/
char s[10005];
int main(){
	int a[4]={0},len,i,j;
	char ch[4]={'G','P','L','T'};
	scanf("%s",&s);
	len=strlen(s);
	for(i=0;i<len;i++){
		if(s[i]=='G' || s[i]=='g') a[0]++;
		if(s[i]=='P' || s[i]=='p') a[1]++;
		if(s[i]=='L' || s[i]=='l') a[2]++;
		if(s[i]=='T' || s[i]=='t') a[3]++;
	}
	int max=a[0];
	for(i=0;i<4;i++){
		if(a[i]>max) max=a[i];
	}
	for(i=0;i<max;i++){
		for(j=0;j<4;j++){
			if(a[j]==0) continue;
			else{
				printf("%c",ch[j]);
				a[j]--;
			}
		}
	}
    return 0;
}
