#include<stdio.h>
int main(){
	int n,i,a[100],count=0,sum=0;
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d",&a[i]);
		sum+=a[i];
	}
	int aver=sum/n;
	for(i=0;i<n;i++){
		a[i]-=aver;//差值数组 如[0,-1,-2,3]，计算每个值为0的移动次数即可，负数合法
	}
	for(i=0;i<n-1;i++){
		if(a[i]!=0){
			a[i+1]+=a[i];
			count++;
			a[i]=0;
		}
	}
	printf("%d",count);
	return 0;
}
