#include<stdio.h>
#include<math.h>
//求2^x+3^x-4^x=0在区间[1,2]内的根，输出后10位  输入区间值 
double f(double x){
	return pow(0.5,x)+pow(0.75,x)-1;
}
int main(){
	double a,b,x;
	scanf("%lf %lf",&a,&b);
	x=(a+b)/2;
	while(fabs(b-a)>1e-4 || fabs(f(x))>1e-4){
		if(f(a)*f(x)<0) b=x;
		if(f(b)*f(x)<0) a=x;
		x=(a+b)/2;
	}//浮点数求绝对值要用fabs 
	printf("%.10lf",x);
	return 0;
}
