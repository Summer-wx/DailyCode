#include<stdio.h>
double wait(int *a,int b){
	int i,j;
	double sum=0;
	for(i=0;i<b;i++){
		for(j=0;j<i;j++){
			sum+=a[j];
		}
		sum+=a[i];
	}
	return sum;
}
//第n个人等待的时间等于T1+...+Tn 
//要一直到这个人接完水才完成等待，接水的时间也是在等待 
int main(){
	int n,i,j,T[1000],k[1000];//k内为数组的下标 
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d",&T[i]);
		k[i]=i+1;
	}
	int x;
	for(i=0;i<n-1;i++){
		for(j=i+1;j<n;j++){
			if(T[i]>T[j]){
				x=T[i];
				T[i]=T[j];
				T[j]=x;
				x=k[i];
				k[i]=k[j];
				k[j]=x;
			}
		}
	}
	for(i=0;i<n;i++){
		printf("%d ",k[i]);
	}
	printf("\n%.2f",wait(T,n)/n);
	return 0;
}
