#include<stdio.h>
int a2(int n){
	int sum=0;
	while(n){
		sum+=n%2;
		n/=2;
	}
	return sum;
}
int a4(int n){
	int sum=0;
	while(n){
		sum+=n%4;
		n/=4;
	}
	return sum;	
}
int main(){
	int i,count=0;
	for(i=1;i<=2024;i++){
		if(a2(i)==a4(i)) count++;
	}
	printf("%d",count);
	return 0;
}
