#include<stdio.h>
void panduan(int year,int month,int day){
	if(year<1900 || year>2026 || month>12 || month<1 || day<1 || day>31) printf("输入有误");
	if(year%400==0 || year%4==0){
		if(month==2){
			if(day>29) printf("输入有误");
		}
		if(month==1||month==3||month==5||month==7||month==8||month==10||month==12){
			if(day>31) printf("输入有误");
		}
		else{
			if(day>30) printf("输入有误");
		}
	}
	else{
		if(month==2){
			if(day>28) printf("输入有误");
		}
		if(month==1||month==3||month==5||month==7||month==8||month==10||month==12){
			if(day>31) printf("输入有误");
		}
		else{
			if(day>30) printf("输入有误");
		}		
	}
	if(month<10){
		if(day<10) printf("%d-0%d-0%d",year,month,day);
		else printf("%d-0%d-%d",year,month,day);
	}
	else{
		if(day<10) printf("%d-%d-0%d",year,month,day);
		else printf("%d-%d-%d",year,month,day);
	}
}
int main(){
	int year,month,day;
	scanf("%d-%d-%d",&month,&day,&year);
	panduan(year,month,day);
    return 0;
}
